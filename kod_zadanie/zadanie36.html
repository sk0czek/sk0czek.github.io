#include<iostream>

            using namespace std;
            
            int main(){
                int a[8][8];
                for(int i=0;i< 8;i++){
                    for(int j=0;j< 8;j++){
                        if(i==j)
                            a[i][j]=9;
                        else
                            a[i][j]=1;
                    }
                }
                for(int x=0;x< 8;x++){
                    for(int y=0;y< 8;y++){
                        cout << a[x][y] << " ";
                    }
                    cout << endl;
                }
            
              return 0;
            }
