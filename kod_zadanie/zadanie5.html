#include<iostream>

        using namespace std;
        
        int main(){
            int a_sko, b_sko;
        
            cout << "Podaj a: "; cin >> a_sko; cout << endl;
            cout << "Podaj b: "; cin >> b_sko; cout << endl;
        
            int suma = a_sko+b_sko, roznica = a_sko-b_sko, iloczyn = a_sko*b_sko, reszta = a_sko%b_sko;
            float iloraz=1.0*a_sko/b_sko;
        
            cout << "Suma = " << suma << endl;
            cout << "Roznica = " << roznica << endl;
            cout << "Iloczyn = " << iloczyn << endl;
            cout << "Iloraz = " << iloraz << endl;
            cout << "Reszta z dzielenia = " << reszta << endl;
        
            return 0;
        }
