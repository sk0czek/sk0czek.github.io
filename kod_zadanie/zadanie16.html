#include<iostream>

        using namespace std;
        
        int nwd(int n, int m){
            int r;
            while(m!=0){
                r=(n%m);
                n=m;
                m=r;
            }
            return n;
        }
        
        int main() {
            int m, n;
        
            cout << "Podaj pierwsza liczbe: "; cin >> m; cout << endl;
            cout << "Podaj druga liczbe: "; cin >> n; cout << endl;
        
            int NWD = nwd(m, n);
        
            cout << "Szukanym dzielnikiem jest: " << nwd(m, n) << endl;
        
            return 0;
        }
